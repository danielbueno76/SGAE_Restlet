010public class URLUtils {
011    /**
012     * Método para extraer el último segmento del URL
013     * (extraído de <a href="http://stackoverflow.com/questions/4050087/how-to-obtain-the-last-path-segment-of-an-uri">
014     * http://stackoverflow.com/questions/4050087/how-to-obtain-the-last-path-segment-of-an-uri</a>)
015     * 
016     * <br>
017     * 
018     * Explanation of 
019     * <code>return url.replaceFirst(".&lowast;/([^/?]+).&lowast;", "$1");
020     * </code> :
021     * <ul>
022     * <li><code>.&lowast;/</code> find anything up to the last / character</li>
023     * 
024     * <li><code>([^/?]+) </code> find (and capture) all following characters 
025     * up to the next / or ? the + makes sure that at least 1 character is 
026     * matched</li>
027     * <li><code>.&lowast;</code> find all following characters</li>
028     * <li><code>$1</code> this variable references the saved second group 
029     * from above
030     *    I.e. the entire string is replaces with just the portion
031     *    captured by the parentheses above</li>
032     * </ul>
033     * @param url URL que se desea procesar
034     * @return el último segmento del URL (sin consultas)
035     */
036    public static String getLastPathFromUrl(final String url){
037        if (url == null)
038            return null;
039        // return url.replaceFirst("[^?]*/(.*?)(?:\\?.*)","$1);" <-- incorrect
040        return url.replaceFirst(".*/([^/?]+).*", "$1");
041    }
042}
