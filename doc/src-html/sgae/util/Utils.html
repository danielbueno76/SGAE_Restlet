001package sgae.util;
002
003import java.text.ParseException;
004
005public class Utils {
006        // From
007        // https://codereview.stackexchange.com/questions/69827/isnullorwhitespace-check-for-java
008        /**
009         * Función que comprueba si una cadena está vacía, contiene sólo espacios o es un puntero null
010         * @param value cadena a comprobar
011         * @return true si la cadena está vacía, contiene sólo espacios o es un puntero null
012         */
013        public static boolean isStringNullOrEmptyOrWhitespace(CharSequence value) {
014                if (value == null) {
015                        return true;
016                }
017
018                if (value.toString().equals("")) {
019                        return true;
020                }
021
022                /*
023                 * From
024                 * https://stackoverflow.com/questions/3247067/how-do-i-check-that-a-
025                 * java-string-is-not-all-whitespaces trim() and other mentioned regular
026                 * expression do not work for all types of whitespaces i.e: Unicode
027                 * Character 'LINE SEPARATOR'
028                 * http://www.fileformat.info/info/unicode/char/2028/index.htm Java
029                 * functions Character.isWhitespace() covers all situations.
030                 */
031                for (int i = 0; i < value.length(); i++) {
032                        if (!Character.isWhitespace(value.charAt(i))) {
033                                return false;
034                        }
035                }
036
037                return true;
038        }
039
040        /**
041         * Función que comprueba si una cadena está vacía, contiene sólo espacios o es un puntero null y lanza 
042         * una excepción ParseExeption en este caso (en caso contrario, devuelve elmismo valor de entrada)
043         * @param value cadena de texto a comprobar
044         * @param errorMsg cadena con el mensaje específico de error que se asociará a la excepción ParseException
045         * @return el valor de la cadena de entrada si contiene algún caracter distinto de espacio, tabulador o retorno de línea
046         * @throws ParseException si la cadena contenida en el parámetro <i>value</i>está vacía, contiene sólo espacios o es un puntero null
047         */
048        public static String testStringNullOrEmptyOrWhitespaceAndSet(String value, String errorMsg) throws ParseException {
049                if (isStringNullOrEmptyOrWhitespace(value)) {
050                        throw new ParseException(errorMsg, 0);
051                }
052                return value;
053        }
054}
