045        /** 
046         * Metodo get en formato texto plano sobre grupo
047         * @return result datos solicitados en texto plano
048         */
049        @Get("txt")
050        public String represent(){
051                
052                String DescripcionGrupo=null;
053                int numerodejemplaresvendidos=0;
054                try {
055                        //Almacenamos en la variable de tipo String la información completa del grupo y el URI relativo para el siguiente elemento. Luego este contenido se muestra.
056                        DescripcionGrupo= controladorGruposMusicales.verGrupoMusical(this.grupoID);
057                        //Obtenemos el umero de ejemplares vendidos
058                        for (Album album: controladorGruposMusicales.recuperarAlbumes(grupoID)) {               
059                                numerodejemplaresvendidos+=album.getEjemplaresVendidos();
060                        }
061                        DescripcionGrupo=DescripcionGrupo+"\tNúmero de ejemplares vendidos: "+numerodejemplaresvendidos+"\tURI: albumes/ \t o \t miembros/";
062                }
063                catch(ExcepcionGruposMusicales ex) {
064                        System.out.println("ExcepcionGruposMusicales No existe el grupo");//No existe el grupo solicitado
065                throw new ResourceException(Status.CLIENT_ERROR_NOT_FOUND,ex.getCausaFallo());//Se devuelve un error --> Recurso no encontrado.
066                }
067                
068                return DescripcionGrupo;//Devolvemos la variable de tipo String con todos los datos.
069        }

072        /** 
073         * Metodo get en formato xml sobre grupo
074         * @return result datos solicitados en xml
075         */
076        @Get("xml")
077        public JaxbRepresentation<sgae.util.generated.Grupo> toXml() {
078                JaxbRepresentation <sgae.util.generated.Grupo> result = null;
079                int numerodejemplaresvendidos=0;
080                try {
081                        //Recuperamos el grupo musical especifico
082                        GrupoMusical g = controladorGruposMusicales.recuperarGrupoMusical(this.grupoID);
083                        //Obtenemos el umero de ejemplares vendidos
084                        for (Album album: controladorGruposMusicales.recuperarAlbumes(grupoID)) {               
085                                numerodejemplaresvendidos+=album.getEjemplaresVendidos();
086                        }
087                        //Añadimos al objeto grupo la información obtenida.
088                        sgae.util.generated.Grupo grupoInfo = new sgae.util.generated.Grupo();  
089                        grupoInfo.setCIF(g.getCif());
090                        grupoInfo.setNombre(g.getNombre());
091                        grupoInfo.setFechacreacion(g.getFechaCreacion());
092                        grupoInfo.setEjemplaresvendidos(String.valueOf(numerodejemplaresvendidos));
093                        //Creamos dos objetos de tipo link que son las dos colecciones que cuelgan.
094                        Link link1 = new Link();
095                        link1.setHref("albumes/");
096                        link1.setTitle("Albumes");
097                        link1.setType("simple");
098                        grupoInfo.setUri1(link1);
099                        Link link2 = new Link();
100                        link2.setHref("miembros");
101                        link2.setTitle("Miembros");
102                        link2.setType("simple");
103                        grupoInfo.setUri2(link2);
104                        //Se crea el objeto JaxbRepresentation result con el formato especificado en la teoría
105                        result = new JaxbRepresentation<sgae.util.generated.Grupo> (grupoInfo);
106                        result.setFormattedOutput(true);
107                }catch(ExcepcionGruposMusicales ex) {
108                        System.out.println("ExcepcionGruposMusicales No existe el grupo");//No existe el grupo solicitado
109                        throw new ResourceException(Status.CLIENT_ERROR_NOT_FOUND,ex.getCausaFallo());//Se devuelve un error --> Recurso no encontrado.
110                }
111                        return result;//Devolvemos la variable de tipo JaxbRepresentation con todos los datos.
112
113        }

115        /** 
116         * Metodo put para crear un grupo o añadir y eliminar miembros.
117         * @param datos Variable en la cual recibimos los datos enviados por el cliente para modificar el album.
118         * @throws ParseException Excepcion producida cuando el cliente envía los datos para crear o  modificar de manera errónea.
119         * @throws ExcepcionGruposMusicales No existe el grupo solicitado.
120         * @return result datos solicitados en el formato solicitado
121         */
122        @Put("form-data")
123        public Representation annadirGrupo (Representation datos){
124                Representation result = null;//En esta variable se almacenará el contenido que devolveremos en modo Representation.
125                //Capturamos los datos de la cabecera
126                Form form = new Form(datos);    
127                //Almacenamos en variables locales los datos necesarios.
128                String CIF= this.grupoID;
129                String DNI= form.getValues("DNI");
130                //Almacenamos en variables locales los datos necesarios para crear o modificar un grupo.
131                String Nombre= form.getFirstValue("NOMBRE");
132                String FechaCreacion= form.getFirstValue("FECHACREACION");
133//              //Mostramos los datos almacenados por si acaso.
134//              System.out.println("CIF: " + CIF );
135//              System.out.println("Nombre: " + Nombre);
136//              System.out.println("Fecha de creacion: " + FechaCreacion);
137                //Queremos crear un grupo y no añadir miembros
138                if (DNI==null &&Nombre!=null &&FechaCreacion!=null) {   
139                        
140                        try {
141                                //Creamos un grupo
142                                controladorGruposMusicales.crearGrupoMusical(CIF, Nombre, FechaCreacion);
143                                //Almacenamos la información que vamos a mostrar al crear el grupo.
144                                result =  new StringRepresentation("CIF: " + CIF +"\tNombre: " + Nombre+"\tFecha de creacion: " + FechaCreacion,   MediaType.APPLICATION_WWW_FORM);
145                                getResponse().setStatus(Status.SUCCESS_CREATED, "Se ha creado con exito");
146                        } 
147                        //Controlamos las posibles excepciones que se pueden producir
148                        catch (ExcepcionGruposMusicales ex){
149                                System.out.println("ExcepcionGruposMusicales crearGrupo");//No existe el grupo solicitado
150                                throw new ResourceException(Status.CLIENT_ERROR_NOT_FOUND, ex.getCausaFallo());//Se devuelve un error --> Recurso no encontrado.
151                                
152                        }catch (ParseException ax) {
153                                System.out.println("ParseException creargrupo");
154                                throw new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST, "Te falta un parámetro o has enviado un parámetro con una sintaxis errónea");//Se devuelve un error --> Los datos enviados por el usuario han sido enviado de manera errónea.
155                        }
156                }
157                //Queremos crear un grupo y añadir miembros
158                else if (DNI!=null &&Nombre!=null &&FechaCreacion!=null) {
159                        String[] arrayDNIs = DNI.split(",");//Dividimos la ristra de DNIS, en varios String.
160                        int i=0;
161                        String listamiembros = "";
162                        try {
163                                //Creamos un grupo
164                                controladorGruposMusicales.crearGrupoMusical(CIF, Nombre, FechaCreacion);
165                                //añadimos los miembros
166                                for(i=0;i<arrayDNIs.length;i++) {
167                                        //Vamos añadiendo en cada iteración del bucle los nuevos miembros
168                                        controladorGruposMusicales.anadirMiembro(CIF, arrayDNIs[i]);
169                                        listamiembros=listamiembros+arrayDNIs[i]+"\t";
170                                }
171                                //Almacenamos la información que vamos a mostrar al crear el grupo.
172                                result =  new StringRepresentation("CIF: " + CIF +"\tNombre: " + Nombre+"\tFecha de creacion: " + FechaCreacion +"\tMiembros añadidos: "+listamiembros,   MediaType.APPLICATION_WWW_FORM);
173                                getResponse().setStatus(Status.SUCCESS_CREATED, "Se ha creado con exito");
174                        } 
175                        //Controlamos las posibles excepciones que se pueden producir
176                        catch (ExcepcionGruposMusicales ex){
177                                System.out.println("ExcepcionGruposMusicales crearGrupo");//No existe el grupo solicitado
178                                throw new ResourceException(Status.CLIENT_ERROR_NOT_FOUND, ex.getCausaFallo());//Se devuelve un error --> Recurso no encontrado.
179                                
180                        }catch (ParseException ax) {
181                                System.out.println("ParseException creargrupo");
182                                throw new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST, "Te falta un parámetro o has enviado un parámetro con una sintaxis errónea");//Se devuelve un error --> Los datos enviados por el usuario han sido enviado de manera errónea.
183                        } catch (ExcepcionPersonas e) {
184                                System.out.println("ExcepcionPersonas añadir Miembro");//No existe la persona solicitado
185                                 throw new ResourceException(Status.CLIENT_ERROR_NOT_FOUND, e.getCausaFallo());//Se devuelve un error --> Recurso no encontrado.
186                        
187                        }
188                        
189                }
190                //añadir o eliminar un miembro.
191                else {
192                        //Ejemplo: DNI=0000000A, 1111111A, 0000000B
193                        String[] arrayDNIs = DNI.split(",");//Dividimos la ristra de DNIS, en varios String.
194                        String[] arraymiembros= null;//Dividimos la lista de miembros en un array de strings.
195                        //Mostramos los DNIs para ver que estan bien
196                        System.out.print("DNIs: "); 
197                        int i=0;
198                        int j=0;
199                        int controlmiembrosannadir=0;//Esta variable auxiliar nos ayudará a saber si un dni de los que le pasamos es un miembro actual
200                        int controlmiembrosborrar=0;//Esta variable auxiliar nos ayudará a saber si borramos un miembro
201                        for(i=0;i<arrayDNIs.length;i++) {
202                                System.out.print(arrayDNIs[i] + ",\t");//Mostrar DNIs
203                        }
204
205                        
206                        try {
207                                StringBuilder result2 = new StringBuilder();//Creamos esta variable que almacenará la información de tipo StringBuilder
208                                //Significa que no existen miembros actuales para ese grupo.
209                                if (controladorGruposMusicales.recuperarMiembros(grupoID).isEmpty()) {
210                                        
211                                        for(i=0;i<arrayDNIs.length;i++) {
212                                                //Vamos añadiendo en cada iteración del bucle los nuevos miembros
213                                                controladorGruposMusicales.anadirMiembro(CIF, arrayDNIs[i]);
214                                                result2.append("Se añade un miembro con DNI: " + arrayDNIs[i] +" al grupo con CIF: " + CIF).append('\n');
215                                        }
216                                        result = new StringRepresentation(result2.toString());//Devolvemos el StringRepresentation con toda la información.
217                                        getResponse().setStatus(Status.SUCCESS_CREATED, "Se ha añadido con exito");
218                                }
219
220                                        //En el caso de que haya un miembro solo y se quiera eliminarlo, se entra aquí. Se le pasa DNI= . Es decir campo vacío.
221                                else if(controladorGruposMusicales.recuperarMiembros(grupoID).size()==1 && arrayDNIs[0].trim().length()==0) {
222                                                //significa que quieres borrar el miembro que hay
223                                                for (Persona persona: controladorGruposMusicales.recuperarMiembros(grupoID)) {
224                                                        //Significa que hay que borrar este miembro
225                                                        controladorGruposMusicales.eliminarMiembro(CIF, persona.getDni());
226                                                        //Información que mostramos tras eliminar el miembro.
227                                                        result2.append("Se elimina un miembro con DNI: " + persona.getDni() +" del grupo con CIF: " + CIF).append('\n');
228                                                }
229                                                getResponse().setStatus(Status.SUCCESS_OK, "Se ha borrado con exito");
230                                        }
231                                        //Significa que queremos añadir o borrar un miembro.
232                                else {
233                                                int tamannolistamiembros=controladorGruposMusicales.recuperarMiembros(grupoID).size();
234                                                arraymiembros= new String[tamannolistamiembros];
235                                                //Vamos comparando cada miembro con cada DNI de la lista que pasamos.
236                                                for (Persona persona: controladorGruposMusicales.recuperarMiembros(grupoID)) {
237                                                        arraymiembros[j]=persona.getDni();
238                                                        j=j+1;
239                                                        for(i=0;i<arrayDNIs.length;i++) {
240                                                                //Significa que hay que borrar ese miembro, ya que no se encuentra en la lista proporcionada
241                                                                if(persona.getDni().equals(arrayDNIs[i])==true) {
242                                                                        controlmiembrosborrar=1;//No queremos borrarlo
243                                                                
244                                                                }
245
246                                                        }
247                                                        if ( controlmiembrosborrar==0) {
248                                                                controladorGruposMusicales.eliminarMiembro(CIF, persona.getDni());
249                                                                //Añadimos la información del miembro borrado que mostraremos.
250                                                                result2.append("Se elimina un miembro con DNI: " + persona.getDni() +" del grupo con CIF: " + CIF).append('\n');
251                                                        }
252                                                        controlmiembrosborrar=0;
253                                                }
254                                                
255                                                //Vamos comparando cada miembro con la lista de DNIs que pasamos.
256                                                        for(i=0;i<arrayDNIs.length;i++) {
257                                                                for(j=0;j<arraymiembros.length;j++) {
258                                                                //Significa que ese DNI no está incluido como miembro actual
259                                                                if(arraymiembros[j].equals(arrayDNIs[i])==true) {       
260                                                                        controlmiembrosannadir=1;//Significa que el DNI pertenece a los miembros actuales       
261                                                                }
262
263                                                        }
264                                                                if(controlmiembrosannadir==0) {
265                                                                        //añadimos el miembro
266                                                                        controladorGruposMusicales.anadirMiembro(CIF, arrayDNIs[i]);
267                                                                        //Añadimos la información que mostraremos.
268                                                                        result2.append("Se añade un miembro con DNI: " + arrayDNIs[i] +" al grupo con CIF: " + CIF).append('\n');
269                                                                }
270                                                                //Si uno de los DNi de la lista del cliente, ya está incluido en la lista de miembros, se mantiene.
271                                                                else {
272                                                                        result2.append("El miembro con DNI: " + arrayDNIs[i] +" del grupo con CIF: " + CIF + " ya pertenece a los miembros actuales").append('\n');
273                                                                }
274                                                                controlmiembrosannadir=0;
275                                                                
276                                        }
277
278                                result = new StringRepresentation(result2.toString());  //Devolvemos la variable de tipo Representation con todos los datos.                    
279                                getResponse().setStatus(Status.SUCCESS_OK, "Se han modificado los miembros con exito");
280                                }
281                        }
282                        catch (ExcepcionPersonas ex){
283                                System.out.println("ExcepcionPersonas añadir Miembro");//No existe la persona solicitado
284                                 throw new ResourceException(Status.CLIENT_ERROR_NOT_FOUND, ex.getCausaFallo());//Se devuelve un error --> Recurso no encontrado.
285                        }catch(ExcepcionGruposMusicales e)
286                        {
287                                System.out.println("ExcepcionGruposMusicales Borrar Album");//No existe el grupo solicitado
288                                throw new ResourceException(Status.CLIENT_ERROR_NOT_FOUND, e.getCausaFallo());//Se devuelve un error --> Recurso no encontrado.
289                        }
290
291                }
292
293                
294                return result;//Devolvemos la variable de tipo Representation con todos los datos.
295                
296        }
