001package sgae.servidor.aplicacion;
002import org.restlet.Component;
003import org.restlet.data.Protocol;
004import org.restlet.Client;
005import org.restlet.Context;
006import org.restlet.Server;
007import org.restlet.routing.VirtualHost;
008
009//Componente Restful que contiene la aplicación SGAE.
010public class SGAEcomponent extends Component{
011        
012        public static void main(String[] args) throws Exception {
013                //Cargamos el componente SgaeComponent
014                new SGAEcomponent().start();            
015        }
016        
017        public SGAEcomponent(){
018                // establecemos las propiedades
019                setName("SGAE Server Component");
020                setDescription("Servidor de SGAE para almacenar informacion sobre grupos musicales");
021                setOwner("ptpdx01");
022                setAuthor("RHB y DBP");
023                
024                // Creamos el componente servidor HTTP
025                Server server = new Server(new Context(), Protocol.HTTP, 8111);
026                //Creamos el componente cliente CLAP.
027                Client client = new Client(new Context(), Protocol.CLAP);
028                //Permitimos el rastreo
029                server.getContext().getParameters().set("tracing", "true");
030                getServers().add(server);//Añadimos el conector
031                getClients().add(client);//Añadimos el cliente
032                //Adherimos la aplicación al host por defecto
033                VirtualHost host = getDefaultHost();
034                host.attachDefault(new SGAEaplicacion());       
035                
036        }
037}
