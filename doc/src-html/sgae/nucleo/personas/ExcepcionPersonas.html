001package sgae.nucleo.personas;
002
003/**
004 * Clase de excepción relativa a personas.
005 * @author Eduardo Gómez Sánchez y Manuel Rodríguez Cayetano. ETSIT UVa.
006 * @version 1.0
007 */
008public class ExcepcionPersonas extends Exception {
009        /** Serial ID que debe ponerse */
010        private static final long serialVersionUID = 1L;
011        /** El DNI de la persona sobre el que se ha hecho una operación 
012            fallida*/
013        private String dniPersona;
014        /** El tipo de operación que ha fallado */
015        private String causaFallo;
016        
017        
018        /** 
019         * Constructor con parámetros.
020         * @param dniPersona DNI de la persona
021         * @param causaFallo texto que describe la causa del fallo
022         */
023        public ExcepcionPersonas(String dniPersona, String causaFallo) {
024                super("Persona: " + dniPersona + ", error: " + causaFallo + "\n");
025                this.dniPersona = dniPersona;
026                this.causaFallo = causaFallo;
027        }
028        
029        /**
030         * Método que devuelve el DNI de la persona.
031         * @return la cadena de texto con el DNI de la persona
032         */
033        public String getDniPersona() {
034                return dniPersona;
035        }
036        
037        /**
038         * Método que devuelve la causa del fallo.
039         * @return la cadena de texto con la causa del fallo
040         */
041        public String getCausaFallo() {
042                return causaFallo;
043        }       
044}
