001package sgae.nucleo.gruposMusicales;
002
003import java.text.ParseException;
004import java.text.SimpleDateFormat;
005import java.util.ArrayList;
006import java.util.Date;
007import java.util.HashMap;
008import java.util.List;
009import java.util.Map;
010
011import sgae.util.Utils;
012
013/**
014 * Clase que almacena información sobre los álbumes de un grupo musical.
015 * @author Manuel Rodríguez Cayetano. ETSIT UVa.
016 * @version 1.0
017 */
018public class Album {
019        /** Attributes */
020        private String idAlbum;
021        private String titulo;
022        private Date fechaPublicacion;
023        private int ejemplaresVendidos;
024        
025        /** Associations */
026        
027        /** La lista de pistas, indexada por un identificador Único */
028        private Map<String,Pista> listaPistas;
029        /** Un contador para generar identificadores Únicos de pista */
030        private int ultimaPista;
031        
032        /** 
033         * Constructor con los atributos que se pueden inicializar de partida. 
034         * Ojo, la fecha de publicación se pasa como una cadena con el formato 
035         * dd-MM-yyyy .
036         *
037         * @param idAlbum el identificador del álbum
038         * @param titulo el título del álbum
039         * @param fechaPublicacion la fecha de publicación del álbum, se 
040         * pasa como una cadena dd-MM-yyyy
041         * @param ejemplaresVendidos número de ejempares vendidos del álbum
042         * @throws ParseException si el parámetro <i>fechaPublicacion</i> no tiene 
043         * el formato dd-MM-yyyy
044         * o si los parámetros idAlbum o titulo están vacíos, contienen sólo espacios
045         * o son null, o si el número de ejemplares vendidos es negativo
046         */
047        public Album(String idAlbum, String titulo, String fechaPublicacion, int ejemplaresVendidos) 
048                throws ParseException {
049                super();
050                // Inicializa con valores pasados como parámetros
051                this.idAlbum = Utils.testStringNullOrEmptyOrWhitespaceAndSet(idAlbum, "Campo idAlbum vacío");
052                this.titulo = Utils.testStringNullOrEmptyOrWhitespaceAndSet(titulo, "Campo título vacío");
053                SimpleDateFormat dateFormat = new SimpleDateFormat("dd-MM-yyyy");
054                dateFormat.setLenient(false);
055                if (Utils.isStringNullOrEmptyOrWhitespace(fechaPublicacion)) {
056                        throw new ParseException("Campo fecha de publicación vacío", 0);
057                }
058                this.fechaPublicacion = dateFormat.parse(fechaPublicacion);
059                if (ejemplaresVendidos < 0) {
060                        throw new ParseException("El número de ejemplares vendidos no puede ser negativo", 0);
061                }
062                this.ejemplaresVendidos = ejemplaresVendidos;
063                // Inicializa una lista de pistas vacía y el contador de pistas
064                listaPistas = new HashMap<String,Pista>();
065                ultimaPista = 0;
066        }
067        
068        /**
069         * Método que lee el identificador del álbum.
070         * NOTA: El identificador sólo se puede leer, no escribir.
071         * @return el valor del identificador
072         */
073        public String getId() {
074                return idAlbum;
075        }
076        
077        /**
078         * Método que devuelve el título del álbum.
079         * @return el título del álbum
080         */
081        public String getTitulo() {
082                return titulo;
083        }
084        
085        /**
086         * Método que modifica el título.
087         * @param nuevoTitulo el nuevo título del nombre
088         * @throws ParseException
089         *             si el parámetro nuevoTitulo está vacío, contiene sólo espacios
090         *             o es null
091         */
092        public void setTitulo(String nuevoTitulo) throws ParseException {
093                titulo = Utils.testStringNullOrEmptyOrWhitespaceAndSet(nuevoTitulo, "Campo título vacío");
094        }
095        
096        /** 
097         * Método que devuelve la fecha de publicación como una cadena.
098         * @return la fecha de publicación en formato dd-MM-yyyy
099         */
100        public String getFechaPublicacion() {
101                return new SimpleDateFormat("dd-MM-yyyy").format(fechaPublicacion);
102        }
103        
104        /**
105         * Método que cambia la fecha de publicación a partir de una cadena.
106         * @param nuevaFechaPublicacion la nueva fecha de publicación del álbum
107         * @throws ParseException si el parámetro <i>fechaPublicacion</i> no tiene 
108         * el formato dd-MM-yyyy
109         */
110        public void setFechaPublicacion(String nuevaFechaPublicacion) 
111                throws ParseException {
112                SimpleDateFormat dateFormat = new SimpleDateFormat("dd-MM-yyyy");
113                dateFormat.setLenient(false);
114                if (Utils.isStringNullOrEmptyOrWhitespace(nuevaFechaPublicacion)) {
115                        throw new ParseException("Campo fecha de publicación vacío", 0);
116                }
117                this.fechaPublicacion = dateFormat.parse(nuevaFechaPublicacion);
118        }
119        
120        /**
121         * Método que devuelve el número de ejemplares vendidos del álbum.
122         * @return el número de ejemplares vendidos
123         */
124        public int getEjemplaresVendidos() {
125                return ejemplaresVendidos;
126        }
127        
128        /**
129         * Método que cambia el número de ejemplares vendidos del álbum.
130         * @param nuevosEjemplaresVendidos el nuevo valor de los ejemplares vendidos
131         * @throws ParseException si el número de ejemplares vendidos es negativo
132         */
133        public void setEjemplaresVendidos(int nuevosEjemplaresVendidos) throws ParseException {
134                if (ejemplaresVendidos < 0) {
135                        throw new ParseException("El número de ejemplares vendidos no puede ser negativo", 0);
136                }
137                ejemplaresVendidos = nuevosEjemplaresVendidos;
138        }
139        
140        
141        /**
142         * Método que devuelve una descripción textual breve del álbum.
143         * @return la descripción textual breve del álbum
144         */
145        String verDescripcionBreve() {
146                return "Título: " + titulo + "\n";
147        }
148        
149        /**
150         * Método que devuelve en una única cadena la información completa del álbum.
151         * 
152         * @return la descripción completa del álbum
153         */
154        String verDescripcionCompleta() {
155                return "Id: " + idAlbum + 
156                        "Título: " + titulo +
157                        "Fecha de publicación: " + fechaPublicacion +
158                        "Ejemplares vendidos: " + ejemplaresVendidos + "\n";
159        }
160        
161        /**
162         * Método que añade un pista, y devuelve su identificador único.
163         * @param nombre nombre de la pista a añadir
164         * @param duracion duración de la pista a añadir
165         * @return el identificador único de la pista añadira al álbum
166         * @throws ExcepcionPistas si los parámetros usados para la creación de la pista
167         * no son correctos
168         * @see sgae.nucleo.gruposMusicales.Pista
169         */
170        public String anadirPista(String nombre, int duracion) throws ExcepcionPistas {
171                // Crea un identificador para la pista, formado por una 'p' y un 
172                // número auto-incrementado
173                String idPista = "p" + ultimaPista;
174                // Crea el objeto
175                Pista p = null;
176                try {
177                        p = new Pista(idPista, nombre, duracion);
178                } catch (ParseException e) {
179                        throw new ExcepcionPistas(idPista, e.getLocalizedMessage());
180                }
181                // Colecciona el objeto indexado por el identificador
182                listaPistas.put(idPista, p);
183                // Incrementa el contador
184                ultimaPista++;
185                // Retorna el identificador del objeto reciÃ©n creado
186                return idPista;
187        }
188
189        /**
190         * Método que comprueba si existe una pista identificada por un número único
191         * @param id identificador de la pista
192         * @return objeto del tipo Pista correspondiente al identificador dado
193         * @throws ExcepcionPistas si no existe una pista con un identificador
194         * igual al valor del parámetro <i>id</i>
195         */
196        private Pista comprobarPistaExiste (String id) 
197                throws ExcepcionPistas {
198                Pista pista = listaPistas.get(id);
199                if (pista == null) {
200                        throw new ExcepcionPistas(id,
201                                                           "La pista que ha especificado no existe");
202                }
203                return pista;
204        }
205        
206        /**
207         * Método que devuelve en una lista de cadenas la información de las 
208         * pistas.
209         * @return la lista formada por cadenas de texto, donde cada una 
210         * contiene la descripción de una pista
211         */
212        public List<String> verPistas() {
213                List<String> listado = new ArrayList<String>();
214                
215                // Recorre la lista de pistas
216                for(Pista p : listaPistas.values()) {
217                        // Y a cada una le pide los detalles
218                        listado.add(p.verDescripcionBreve());
219                }
220                return listado;
221        }
222        
223        /**
224         * Método que devuelve las pistas en una lista de objetos 
225         * de tipo Pista.
226         * @return la lista cuyos elementos son objetos del tipo Pista
227         */
228        public List<Pista> recuperarPistas() {
229                return new ArrayList<Pista>(listaPistas.values());
230        }
231        
232        /**
233         * Método que permite ver la descripción textual de una pista de este álbum.
234         * @param idPista el identificador único de la pista a mostrar
235         * @return una cadena con la descripción de la pista
236         * @throws ExcepcionPistas si no existe la pista que se busca
237         */
238        public String verPista(String idPista) throws ExcepcionPistas {
239                // Intenta obtener el objeto
240                return comprobarPistaExiste(idPista).verDescripcionCompleta();
241        }
242        
243        /**
244         * Método que permite obtener el objeto que representa a una pista dada
245         * en este álbum
246         * @param idPista el identificador único de la pista a recuperar
247         * @return un objeto de tipo Pista
248         * @throws ExcepcionPistas si no existe la pista que se busca
249         */
250        public Pista recuperarPista(String idPista) throws ExcepcionPistas {
251                return comprobarPistaExiste(idPista);
252        }
253
254        /**
255         * Comprueba si el álbum tiene una pista con un nombre determinado.
256         * @param nombrePista nombre de la pista a buscar
257         * @return valor booleano <i>true</i> si existe la pista buscada
258         */
259        public boolean tienePista (String nombrePista) {
260                for (Pista p: listaPistas.values()) {
261                        if (p.getNombre().equals(nombrePista)) {
262                                return true;
263                        }
264                }
265                return false;
266        }
267        
268        /**
269         * Método que elimina una pista.
270         * @param idPista el identificador único de la pista a eliminar
271         * @throws ExcepcionPistas si no existe una pista con el idientificador que se 
272         * ha pasado como parámetro
273         */
274        public void eliminarPista(String idPista) throws ExcepcionPistas {
275                // Intenta borrar el objeto
276                comprobarPistaExiste(idPista);
277                listaPistas.remove(idPista);
278        }    
279}
