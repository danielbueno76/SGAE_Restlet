034         * al controlador de personas.

036         */
037        public ControladorGruposMusicales(InterfazControladorPersonas cp) {
038                this.cp = cp;
039                listaGruposMusicales = new HashMap<String,GrupoMusical>();
040        }

042        /**
043         * Método para crear un grupo musical.
044         * @param cif CIF del grupo musical
045         * @param nombre nombre del grupo musical
046         * @param fechaCreacion fecha de creación del grupo musical.
047         * @throws ParseException si la fecha no tiene el formato día-mes-año (dd-MM-yyyy)
048         * @throws ExcepcionGruposMusicales si el grupo muiscal con CIF igual al valor del 
049         * parámetro <i>cif</i> ya existe
050         */
051        public void crearGrupoMusical(String cif, String nombre, String fechaCreacion)
052                throws ParseException, ExcepcionGruposMusicales {
053                // Comprueba si no existe ya
054                if(listaGruposMusicales.containsKey(cif) == false) {
055                        // Crea la instancia
056                        GrupoMusical gm = new GrupoMusical(cif, nombre, fechaCreacion);
057                        // La colecciona, indexada por CIF
058                        listaGruposMusicales.put(cif, gm);
059                } else {
060                        throw new ExcepcionGruposMusicales(cif,
061                                                           "El grupo musical que se ha intentado crear ya existe");
062                }
063        }
064
065        /**
066         * Método que comprueba si existe un grupo musical identificado por un CIF
067         * @param cif CIF del grupo musical
068         * @return objeto del tipo GrupoMusical correspondiente al CIF dado
069         * @throws ExcepcionGruposMusicales si no existe un grupo musical con un CIF
070         * igual al valor del parámetro <i>cif</i>
071         */
072        private GrupoMusical comprobarGrupoMusicalExiste (String cif) 
073                throws ExcepcionGruposMusicales {
074                GrupoMusical grupoMusical = listaGruposMusicales.get(cif);
075                if (grupoMusical == null) {
076                        throw new ExcepcionGruposMusicales(cif,
077                                                           "El grupo musical que ha especificado no existe");
078                }
079                return grupoMusical;            
080        }

082        /**
083         * Método para modificar un grupo musical (salvo su CIF)
084         * @param cif CIF del grupo musical al modificar
085         * @param nombre nuevo nombre del grupo
086         * @param fechaCreacion nueva fecha de creación del grupo
087         * @throws ParseException si la fecha no está en formato día-mes-año (dd-MM-yyyy)
088         * @throws ExcepcionGruposMusicales si no existe un grupo musical con un CIF
089         * igual al valor del parámetro <i>cif</i>
090         */
091        public void modificarGrupoMusical(String cif, String nombre, String fechaCreacion)
092                throws ParseException, ExcepcionGruposMusicales {
093                // Recupera la instancia
094                GrupoMusical gm = comprobarGrupoMusicalExiste(cif);
095                gm.setNombre(nombre);
096                gm.setFechaCreacion(fechaCreacion);
097        }

099        /**
100         * Método que devuelve una lista de cadenas que representan información breve de 
101         * los grupos musicales.
102         * @return una lista de cadenas de texto, cada una contiene la información breve de 
103         * un grupo musical
104         */
105        public List<String> listarGruposMusicales() {
106                List<String> listado = new ArrayList<String>();
107                // Recorre la lista de grupos musicales
108                for(GrupoMusical gm : listaGruposMusicales.values()) {
109                        // A cada una le pide su descripción breve
110                        listado.add(gm.verDescripcionBreve());
111                }
112                return listado;
113        }
