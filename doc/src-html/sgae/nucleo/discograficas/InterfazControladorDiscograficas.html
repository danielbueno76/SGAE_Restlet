001package sgae.nucleo.discograficas;
002
003import java.text.ParseException;
004import java.util.List;
005import sgae.nucleo.personas.Persona;
006
007/**
008 * Interfaz para la clase controlador de discográficas.
009 * 
010 * @author Eduardo Gómez Sánchez y Manuel Rodríguez Cayetano. ETSIT UVa.
011 * @version 1.0
012 */
013public interface InterfazControladorDiscograficas {
014
015        /**
016         * Método que crea una nueva discográfica y la añade a la colección
017         * 
018         * @param cif
019         *            cif de la discográfica
020         * @param nombre
021         *            nombre de la discográfica
022         * @param direccion
023         *            dirección de la discográfica
024         * @throws ExcepcionDiscograficas
025         *             si ya existe una discográfica con un CIF igual al valor del
026         *             parámetro <i>cif</i>
027         * @throws ParseException si los parámetros cif, nombre o dirección están vacíos, contienen sólo espacios o son el puntero null
028         */
029        void crearDiscografica(String cif, String nombre, String direccion) throws ExcepcionDiscograficas, ParseException;
030
031        /**
032         * Método que permite modificar una discográfica, recibiendo todos los
033         * campos (el CIF no puede cambiar).
034         * 
035         * @param cif
036         *            cif de la discográfica
037         * @param nombre
038         *            nombre de la discográfica
039         * @param direccion
040         *            dirección de la discográfica
041         * @throws ExcepcionDiscograficas
042         *             si no existe una discográfica con un CIF igual al valor del
043         *             parámetro <i>cif</i>
044         * @throws ParseException si los parámetros cif, nombre o dirección están vacíos, contienen sólo espacios o son el puntero null
045         */
046        void modificarDiscografica(String cif, String nombre, String direccion) throws ExcepcionDiscograficas, ParseException;
047
048        /**
049         * Método que devuelve una lista de discográficas, con su descripción breve
050         * 
051         * @return lista de cadenas de texto, donde cada una contiene la descripción
052         *         breve de una discográfica
053         */
054        List<String> listarDiscograficas();
055
056        /**
057         * Método que devuelve una lista de objetos que representan a las
058         * discográficas
059         * 
060         * @return lista de objetos de tipo Discografica
061         */
062        List<Discografica> recuperarDiscograficas();
063
064        /**
065         * Método que permite ver los detalles de una discográfica.
066         * 
067         * @param cif
068         *            cif de la discográfica que se desea consultar
069         * @return cadena de texto con la descripción completa de la discográfica
070         * @throws ExcepcionDiscograficas
071         *             si no existe una discográfica con un CIF igual al valor del
072         *             parámetro <i>cif</i>
073         */
074        String verDiscografica(String cif) throws ExcepcionDiscograficas;
075
076        /**
077         * Método que permite recuperar el objeto que representa a una discográfica.
078         * 
079         * @param cif
080         *            cif de la discográfica que se desea consultar
081         * @return objeto de tipo Discografica
082         * @throws ExcepcionDiscograficas
083         *             si no existe una discográfica con un CIF igual al valor del
084         *             parámetro <i>cif</i>
085         */
086        Discografica recuperarDiscografica(String cif) throws ExcepcionDiscograficas;
087
088        /**
089         * Método que permite borrar una discográfica.
090         * 
091         * @param cif
092         *            de la discográfica que se desea borrar
093         * @throws ExcepcionDiscograficas
094         *             si no existe una discográfica con un CIF igual al valor del
095         *             parámetro <i>cif</i>
096         */
097        void borrarDiscografica(String cif) throws ExcepcionDiscograficas;
098
099        /**
100         * Método que permite ver el número de grupos contratados por una
101         * discográfica.
102         * 
103         * @param cif
104         *            cif de la discográfica
105         * @return el número de grupos contratados de la discográfica
106         * @throws ExcepcionDiscograficas
107         *             si no existe una discográfica con un CIF igual al valor del
108         *             parámetro <i>cif</i>
109         */
110        int verNumeroGruposDeDiscografica(String cif) throws ExcepcionDiscograficas;
111
112        /**
113         * Método que permite ver la masa salarial de una discográfica.
114         * 
115         * @param cif
116         *            cif de la discográfica
117         * @return la masa salarial de la discográfica
118         * @throws ExcepcionDiscograficas
119         *             si no existe una discográfica con un CIF igual al valor del
120         *             parámetro <i>cif</i>
121         */
122        float verMasaSalarialDeDiscografica(String cif) throws ExcepcionDiscograficas;
123
124        /**
125         * Método que permite contratar a un grupo musical
126         * 
127         * @param cifDiscografica
128         *            CIF de la discográfica
129         * @param cifGrupoMusical
130         *            CIF del grupo musical
131         * @param fechaInicio
132         *            fecha de inicio del contrato
133         * @param fechaFin
134         *            fecha de finalización del contrato
135         * @param sueldo
136         *            sueldo asociado al contrato
137         * @return identificador único del nuevo contrato
138         * @throws ExcepcionContratos
139         *             si el grupo musical cuyo CIF toma el valor del parámetro
140         *             <i>cifGrupoMusical</i> ya está contratado o no existe, o si
141         *             la discográfica con CIF igual al valor del parámetro
142         *             <i>cifDiscografica</i> no existe
143         */
144        String nuevoContrato(String cifDiscografica, String cifGrupoMusical, String fechaInicio, String fechaFin,
145                        float sueldo) throws ExcepcionContratos, ParseException;
146
147        /**
148         * Método que permite terminar un contrato de una discográfica
149         *  
150         * @param cifDiscografica CIF de la discográfica
151         * @param idContrato identificador  
152         * @throws ExcepcionContratos
153         *             si la discográfica con CIF igual al valor del parámetro
154         *             <i>cifDiscografica</i> no existe o no tiene asociado el contrato
155         *             con identificadro <i>idContrato</i>
156         */
157        void terminarContrato(String cifDiscografica, String idContrato) throws ExcepcionContratos;
158
159        /**
160         * Método que permite recuperar un contrato de una discográfica
161         *  
162         * @param cifDiscografica CIF de la discográfica
163         * @param idContrato identificador  del contrato
164         * @return un objeto de la clase Contrato
165         * @throws ExcepcionContratos
166         *             si la discográfica con CIF igual al valor del parámetro
167         *             <i>cifDiscografica</i> no existe o si el contrato
168         *             con identificador <i>idContrato</i> no existe
169         */
170        Contrato recuperarContrato(String cifDiscografica, String idContrato) throws ExcepcionContratos;
171
172        /**
173         * Método que permite recuperar una descripción textual de un contrato de una 
174         * discográfica.
175         *  
176         * @param cifDiscografica CIF de la discográfica
177         * @param idContrato identificador del contrato  
178         * @return una representación textual de un Contrato
179         * @throws ExcepcionContratos
180         *             si la discográfica con CIF igual al valor del parámetro
181         *             <i>cifDiscografica</i> no existe o si el contrato
182         *             con identificador <i>idContrato</i> no existe
183         */
184        String verContrato(String cifDiscografica, String idContrato) throws ExcepcionContratos;
185
186        /**
187         * Método que devuelve una descripción del contrato actual de un grupo
188         * musical.
189         * 
190         * @param cifGrupoMusical
191         *            CIF del grupo musical
192         * @return descripción textual del contrato
193         * @throws ExcepcionContratos
194         *             si no hay un contrato activo para el grupo musical con CIF
195         *             igual al valor del parámetro <i>cifGrupoMusical</i> o dicho
196         *             grupo no existe
197         */
198        String verContratoActualDeGrupoMusical(String cifGrupoMusical) throws ExcepcionContratos;
199
200        /**
201         * Método que devuelve el contrato actual de un grupo musical.
202         * 
203         * @param cifGrupoMusical
204         *            CIF del grupo musical
205         * @return el contrato
206         * @throws ExcepcionContratos
207         *             si no hay un contrato activo para el grupo musical con CIF
208         *             igual al valor del parámetro <i>cifGrupoMusical</i> o dicho
209         *             grupo no existe
210         */
211        Contrato recuperarContratoActualDeGrupoMusical(String cifGrupoMusical) throws ExcepcionContratos;
212
213        /**
214         * Método que devuelve una descripción de los contratos anteriores de un
215         * grupo musical
216         * 
217         * @param cifGrupoMusical
218         *            CIF del grupo musical
219         * @return lista de cadenas de texto, donde cada una contiene la descripción
220         *         de un contrato del grupo musical (puede estar vacía)
221         */
222        List<String> verContratosAnterioresDeGrupoMusical(String cifGrupoMusical);
223
224        /**
225         * Método que devuelve una lista de los contratos anteriores de un
226         * grupo musical
227         * 
228         * @param cifGrupoMusical
229         *            CIF del grupo musical
230         * @return lista de contratos anteriores del grupo musical (puede estar vacía)
231         */
232        List<Contrato> recuperarContratosAnterioresDeGrupoMusical(String cifGrupoMusical);
233
234        /**
235         * Método que devuelve una descripción de los contratos vigentes de una
236         * compañía discográfica
237         * 
238         * @param cifDiscografica CIF de la compañía discográfica
239         * @return lista de cadenas de texto, donde cada una contiene la descripción
240         *         de un contrato
241         * @throws ExcepcionDiscograficas
242         *             si la compañía discográfica con CIF igual al valor del parámetro
243         *             <i>cifDiscografica</i> no existe
244         */
245        List<String> verContratosVigentesDeDiscografica(String cifDiscografica) throws ExcepcionDiscograficas;
246
247        /**
248         * Método que devuelve una lista de los contratos vigentes de una
249         * cimpañía discográfica
250         * 
251         * @param cifDiscografica CIF de la compañía discográfica
252         * @return lista de contratos vigentes de la compañia discográfica
253         * @throws ExcepcionDiscograficas
254         *             si la compañia discográfica con CIF igual al valor del parámetro
255         *             <i>cifDiscografica</i> no existe
256         */
257        List<Contrato> recuperarContratosVigentesDeDiscografica(String cifDiscografica) throws ExcepcionDiscograficas;
258
259        /**
260         * Método que devuelve una descripción de los contratos anteriores de una
261         * compañía discográfica
262         * 
263         * @param cifDiscografica CIF de la compañía discográfica
264         * @return lista de cadenas de texto, donde cada una contiene la descripción
265         *         de un contrato
266         * @throws ExcepcionDiscograficas
267         *             si la compañía discográfica con CIF igual al valor del parámetro
268         *             <i>cifDiscografica</i> no existe
269         */
270        List<String> verContratosAnterioresDeDiscografica(String cifDiscografica) throws ExcepcionDiscograficas;
271
272        /**
273         * Método que devuelve una lista de los contratos anteriores de una
274         * compañía discográfica
275         * 
276         * @param cifDiscografica CIF de la compañía discográfica
277         * @return lista de contratos anteriores de la compañia discográfica
278         * @throws ExcepcionDiscograficas
279         *             si la compañia discográfica con CIF igual al valor del parámetro
280         *             <i>cifDiscografica</i> no existe
281         */
282        List<Contrato> recuperarContratosAnterioresDeDiscografica(String cifDiscografica) throws ExcepcionDiscograficas;
283
284        /**
285         * Método que establece el director de una discográfica.
286         * @param cif cif de la discográfica
287         * @param director objeto de la clase Persona que será el nuevo director
288         * @throws ExcepcionDiscograficas si no existe una discográfica con un CIF igual al valor del parámetro <i>cif</i>
289         */     
290        public void nuevoDirector(String cif, Persona director) throws ExcepcionDiscograficas;
291        
292        /**
293         * Método que permite recuperar la información del director de la discográfica.
294         * @param cif cif de la discográfica
295         * @return un objeto de la clase Persona con los datos del director
296         * @throws ExcepcionDiscograficas si no existe una discográfica con un CIF igual al valor del parámetro <i>cif</i>
297         */
298        public Persona recuperarDirector(String cif) throws ExcepcionDiscograficas;
299        
300        /**
301         * Método para borrar el director de una discográfica.
302         * @param cif cif de la discográfica
303         * @throws ExcepcionDiscograficas si no existe una discográfica con un CIF igual al valor del parámetro <i>cif</i>
304         */
305        public void borrarDirector(String cif) throws ExcepcionDiscograficas;
306        
307        /** 
308         * Método para listar todos los directores anteriores de una discográfica.
309         * @param cif cif de la discográfica
310         * @return lista de objetos de la clase @see Persona
311         * @throws ExcepcionDiscograficas si no existe una discográfica con un CIF igual al valor del parámetro <i>cif</i>
312         */
313        public List<Persona> recuperarDirectoresAnteriores(String cif) throws ExcepcionDiscograficas;
314}
