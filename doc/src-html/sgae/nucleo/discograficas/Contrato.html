001package sgae.nucleo.discograficas;
002import java.text.ParseException;
003
004import sgae.nucleo.gruposMusicales.GrupoMusical;
005import sgae.util.Utils;
006
007/**
008 * Clase que recoge las características de un contrato y los métodos para 
009 * consultar o modificar dichas características.
010 * @author Eduardo Gómez Sánchez y Manuel Rodríguez Cayetano. ETSIT UVa.
011 * @version 1.0
012 */
013public class Contrato {
014    /** Un identificador único */
015    private String id;
016    /** El grupo musica que es contratado */
017    private GrupoMusical grupoMusical;
018    /** Fecha de inicio */
019    private String fechaInicio;
020    /** Fecha de finalización */
021    private String fechaFin;
022    /** El sueldo que se percibe */
023    private float sueldo;
024    /** Si es false, éste fue un contrato pasado pero está rescindido */
025    private boolean activo;
026
027    /**
028     * Constructor que recibe como parámetros la discográfica, el grupo musical, 
029     * el sueldo, la fecha de inicio del contrato y la fecha de finalización.
030     * Por defecto el contrato se crea como activo.
031     * @param id un identificador único
032     * @param grupoMusical el grupo musical que es contratado
033     * @param fechaInicio fecha de inicio del contrato
034     * @param fechaFin fecha de finalización del contrato
035     * @param sueldo el sueldo que se percibe
036     * @throws ParseException si el parámetro id está vacío, contiene sólo espacios o es el puntero null, o 
037     * si el sueldo es un número negativo o si los parámetros fechaInicio y fechaFin no tienen el formato dd-MM-yyyy
038     */
039    public Contrato(String id, GrupoMusical grupoMusical,
040                    String fechaInicio, String fechaFin, float sueldo) throws ParseException {
041        super();
042        // Inicializa según los parámetros
043        this.id = Utils.testStringNullOrEmptyOrWhitespaceAndSet(id, "Campo id vacío");
044        this.grupoMusical = grupoMusical;
045        this.fechaInicio = Utils.testStringNullOrEmptyOrWhitespaceAndSet(fechaInicio, "Campo fecha inicio vacío");
046        this.fechaFin = Utils.testStringNullOrEmptyOrWhitespaceAndSet(fechaFin, "Campo fecha fin vacío");
047        if (sueldo < 0) {
048                throw new ParseException("El sueldo no puede ser negativo", 0);
049        }
050        this.sueldo = sueldo;
051        // Por defecto un contrato se crea como activo
052        this.activo = true;
053        // Decimos al  grupo musical que está contratado
054        grupoMusical.contrata();
055    }
056
057    /**
058     * Método que devuelve el id del contrato.
059     * @return el identificador del contrato
060     */
061    public String getId() {
062        return id;
063    }
064
065    /**
066     * Método que devuelve el sueldo del contrato.
067     * @return el sueldo del contrato
068     */
069    public float getSueldo() {
070        return sueldo;
071    }
072    
073    /**
074     * Método que modifica el sueldo del contrato.
075     * @param sueldo el nuevo sueldo del contrato
076     * @throws ParseException si el sueldo es un número negativo
077     */
078    public void setSueldo(float sueldo) throws ParseException {
079        if (sueldo < 0) {
080                throw new ParseException("El sueldo no puede ser negativo", 0);
081        }
082        this.sueldo = sueldo;
083    }
084
085   /**
086     * Método que devuelve la fecha de inicio del contrato.
087     * @return texto con la fecha de inicio del contrato
088     */
089    public String getFechaInicio() {
090        return fechaInicio;
091    }
092
093    /**
094     * Método que modifica la fecha de inicio del contrato.
095     * @param fechaInicio la nueva fecha de inicio del contrato
096     * @throws si el valor de la fecha no tiene el formato dd-MM-yyyy
097     */
098    public void setFechaInicio(String fechaInicio) throws ParseException {
099        this.fechaInicio = Utils.testStringNullOrEmptyOrWhitespaceAndSet(fechaInicio, "Campo fecha inicio vacío");
100    }
101
102    /**
103     * Método que devuelve la fecha de finalización del contrato.
104     * @return texto con la fecha de finalización del contrato
105     */
106    public String getFechaFinalizacion() {
107        return fechaFin;
108    }
109
110    /**
111     * Método que modifica la fecha de inicio del contrato.
112     * @param fechaFinalizacion la nueva fecha de finalización del contrato
113     * @throws si el valor de la fecha no tiene el formato dd-MM-yyyy
114     */
115    public void setFechaFinalizacion(String fechaFinalizacion) throws ParseException {
116        this.fechaFin = Utils.testStringNullOrEmptyOrWhitespaceAndSet(fechaFinalizacion, "Campo fecha fin vacío");
117    }
118
119   /** 
120     * Método que informa de si el contrato está activo.
121     * @return true si el contrato está activo
122     */
123    public boolean isActivo() {
124        return activo;
125    }
126
127    /**
128     * Método que termina el contrato.
129     */
130    public void terminar() {
131        // Decimos al grupoMusical que está parado
132        grupoMusical.despide();
133        // Cambiamos el estado del contrato
134        activo = false;
135    }
136    
137    /**
138     * Método que devuelve en una única cadena toda la información de un 
139     * contrato.
140     * @return la cadena de texto con la información completa del contrato
141     */
142    public String verDetalles() {
143        return "Contrato con grupo musical con CIF " + grupoMusical.getCif() 
144            + " por " + String.format("%.2f", sueldo) + " euros."
145            + " Situación: " + (activo ? "EN VIGOR" : "TERMINADO")
146            + "Fecha de inicio: " + fechaInicio
147            + "Fecha de finalización " + fechaFin;
148    }
149
150    /**
151     * Método que comprueba si un contrato tiene a un grupo musical determinado
152     * @param cifGrupoMusical identificador del grupo musical
153     * @return true si el grupo musical está asociada al contrato
154     */
155    public boolean tieneGrupoMusical(String cifGrupoMusical) {
156        return grupoMusical.getCif().equals(cifGrupoMusical);
157    }
158    
159    /**
160     * Método que devuelve el identificador del grupo musical que tiene este contrato
161     * @return el identificador del grupo musical que está asociado al contrato
162     */
163    public String getCifGrupoMusical() {
164        // Le pide al grupo musical su id y lo devuelve
165        return grupoMusical.getCif();
166    }
167}
