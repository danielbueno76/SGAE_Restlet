169        public String nuevoContrato(GrupoMusical grupoMusical, String fechaInicicio, String fechaFin, float sueldo)
170                        throws ExcepcionContratos, ParseException {
171                // Genera un identificador único de contrato
172                String idContrato = "c" + ultimoContrato;
173                // Crea la instancia
174                Contrato c = new Contrato(idContrato, grupoMusical, fechaInicicio, fechaFin, sueldo);
175                // Y la colecciona
176                listaContratosActivos.put(idContrato, c);
177                // Incrementa el identificador de contrato
178                ultimoContrato++;
179                // Incrementa el número de empleados
180                numeroGruposContratados++;
181                // Incrementa la masa salarial
182                masaSalarial += sueldo;
183                // Devuelve el identificador del contrato recién creado
184                return idContrato;
185        }

189     * una discográfica.
190     * @return lista  de cadenas de texto, donde cada una contiene la 
191     * descripción de un contrato de la discográfica
192     */
193    public List<String> verContratosVigentes() {
194        // Se inicializa una lista
195        List<String> listado = new ArrayList<String>();
196
197            // Recorre la lista de contratos terminados
198            for(Contrato c : listaContratosActivos.values()) {

200                    listado.add(c.verDetalles());
201            }
202        
203        // Al terminar se devuelve el listado (que podría estar vacío)
204        return listado;
205    }

246                        throw new ExcepcionContratos(idContrato, "No se ha podido encontrar el contrato que se desea ver.");

361        /**
362         * Método que permite terminar un contrato cuyo identificador se pasa como
363         * parámetro.
364         * 
365         * @param idContrato
366         *            identificador del contrado
367         * @throws ExcepcionContratos
368         *             si el contrato no existe
369         */
370        public void terminarContrato(String idContrato) throws ExcepcionContratos {
371                // Recupera la instancia de contrato
372                Contrato c = listaContratosActivos.get(idContrato);
373                // Si existe ese contrato
374                if (c != null) {
375                        // Delega en él las tareas de que se termine
376                        c.terminar();
377                        // Lo saca del listado de contratos activos
378                        listaContratosActivos.remove(idContrato);
379                        // Y lo añade a la de los terminados
380                        listaContratosTerminados.put(idContrato, c);
381                        // Decrementa el número de empleados
382                        numeroGruposContratados--;
383                        // Decrementa la masa salarial
384                        masaSalarial -= c.getSueldo();
385                } else {
386                        // Si el contrato no existía, lanzamos una excepción
387                        throw new ExcepcionContratos(idContrato, "No se ha podido encontrar el contrato que se desea terminar.");
388                }
389        }

391        /**
392         * Método que establece el director de una discográfica.
393         * @param director objeto de la clase Persona que será el nuevo director
394         */     
395        public void nuevoDirector(Persona director) {
396                if (this.director != null) {
397                        // Si el último director ya está en la lista de directores anteriores, no se añade a ella
398                        if (!listaDirectoresAnteriores.containsKey(this.director.getDni())) {
399                                if (director != null) {
400                                        // Sólo se añade un director a la lista de directores anteriores si 
401                                        // el nuevo director es distinto del director actual
402                                        if (director.getDni().compareTo(this.director.getDni()) != 0) {
403                                                listaDirectoresAnteriores.put(this.director.getDni(), this.director)
404;
405                                        }
406                                } else {
407                                        // Nuevo director vacío
408                                        listaDirectoresAnteriores.put(this.director.getDni(), this.director);
409                                }
410                        }
411                }
412                // Se fija el director: si el nuevo y el actual es el mismo esta asignación 
413                // no tiene efecto; si el nuevo director es null, el efecto es dejar a la 
414                // discográfica sin director
415                this.director = director;
416        }
